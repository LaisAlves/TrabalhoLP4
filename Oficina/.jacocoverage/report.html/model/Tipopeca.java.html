        return idTipopeca;
    }

    public void setIdTipopeca(Integer idTipopeca) {
        this.idTipopeca = idTipopeca;
    }

    public String getNome() {
        return nome;
    }

    public void setNome(String nome) {
        this.nome = nome;
    }

        int hash = 0;

    }

    }

}
